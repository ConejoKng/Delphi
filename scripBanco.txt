CREATE TABLE AGENDAMENTO (
	ID INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL,
	ID_PACIENTE INTEGER,
	"DATA" DATE,
	HORA VARCHAR(5),
	ESPECIALIDADE VARCHAR(25),
	MEDICO VARCHAR(30),
	CONSTRAINT INTEG_2 PRIMARY KEY (ID)
);
CREATE INDEX FK_AGENDAMENTO_PACIENTE ON AGENDAMENTO (ID_PACIENTE);
CREATE UNIQUE INDEX RDB$PRIMARY2 ON AGENDAMENTO (ID);


-- AGENDAMENTO chaves estrangeiras

ALTER TABLE AGENDAMENTO ADD CONSTRAINT FK_AGENDAMENTO_PACIENTE FOREIGN KEY (ID_PACIENTE) 
REFERENCES PACIENTE(ID) ON DELETE CASCADE ON UPDATE CASCADE;

CREATE TABLE PACIENTE (
	ID INTEGER GENERATED BY DEFAULT AS IDENTITY NOT NULL,
	CPF VARCHAR(14),
	NOME VARCHAR(80),
	CELULAR VARCHAR(16),
	DATA_CADASTRO DATE,
	CONSTRAINT INTEG_1 PRIMARY KEY (ID)
);
CREATE UNIQUE INDEX RDB$PRIMARY1 ON PACIENTE (ID);
